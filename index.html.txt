  // Navbar scroll
  window.addEventListener('scroll', () => {
    document.getElementById('navbar').classList.toggle('scrolled', window.scrollY > 20);
  });

  // Goal bar toggle
  function setGoal(el) {
    document.querySelectorAll('.goal-item').forEach(i => i.classList.remove('active'));
    el.classList.add('active');
  }

  // Route accordion
  function toggleRoute(btn) {
    const content = btn.nextElementSibling;
    const isOpen = btn.classList.contains('open');
    document.querySelectorAll('.route-btn').forEach(b => { b.classList.remove('open'); b.nextElementSibling.classList.remove('open'); });
    if (!isOpen) { btn.classList.add('open'); content.classList.add('open'); }
  }

  // Tax calc
  function fmtINR(n) {
    if (n >= 100000) return '₹' + (n/100000).toFixed(1) + 'L';
    if (n >= 1000) return '₹' + Math.round(n/1000) + 'K';
    return '₹' + Math.round(n);
  }

  function updateCalc(val) {
    val = parseInt(val);
    document.getElementById('calcDisplay').textContent = '₹ ' + val.toLocaleString('en-IN');
    document.getElementById('slab1').textContent = fmtINR(val * 0.103);
    document.getElementById('slab2').textContent = fmtINR(val * 0.206);
    document.getElementById('slab3').textContent = fmtINR(val * 0.309);
  }

  // SIP calc
  function updateSip() {
    const sliders = document.querySelectorAll('.sip-field input[type=range]');
    const monthly = parseInt(sliders[0].value);
    const years = parseInt(sliders[1].value);
    const rate = parseFloat(sliders[2].value);
    document.getElementById('sipAmt').textContent = '₹' + monthly.toLocaleString('en-IN');
    document.getElementById('sipYrs').textContent = years + ' yrs';
    document.getElementById('sipRate').textContent = rate + '%';
    const months = years * 12;
    const r = rate / 100 / 12;
    const fv = monthly * ((Math.pow(1+r, months) - 1) / r) * (1+r);
    const invested = monthly * months;
    const returns = fv - invested;
    const fmt = (n) => {
      if (n >= 10000000) return '₹' + (n/10000000).toFixed(2) + 'Cr';
      if (n >= 100000) return '₹' + (n/100000).toFixed(2) + 'L';
      return '₹' + Math.round(n).toLocaleString('en-IN');
    };
    document.getElementById('sipInvested').textContent = fmt(invested);
    document.getElementById('sipReturns').textContent = fmt(returns);
    document.getElementById('sipTotal').textContent = fmt(fv);
  }
  updateSip();

  // Intersection observer for reveal
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(e => { if (e.isIntersecting) { e.target.classList.add('visible'); } });
  }, { threshold: 0.12 });
  document.querySelectorAll('.reveal').forEach(el => observer.observe(el));

  // Form submit

  }

  function toggleMenu() {
    const links = document.querySelector('.nav-links');

    }
  }
